CREATE TABLE IF NOT EXISTS peoples_t (
  id serial primary key
  phone varchar(10) 
  name varchar(50)
  total integer 
)

CREATE TABLE IF NOT EXISTS drinks_t (
  name varchar(50) primary key
  price integer
)

CREATE TABLE IF NOT EXISTS queue_t (
  order_number integer primary key references orders_t(order_number)
)

// should be able to act as queue

  order_number serial primary key
  person_id varchar(10) references peoples_t(id)
